package megamek.server;

import java.util.TimerTask;

public class ConnectionWatchdog extends TimerTask {

    private Server server;

    private int id;

    private int failCount;

    public ConnectionWatchdog(Server server, int id) {
        this.server = server;
        this.id = id;
        failCount = 0;
    }

    @Override
    public void run() {
        if (server.getPlayer(id) != null) {
            // fully connected
            cancel();
            return;
        }
        if (server.getPendingConnection(id) == null) {
            // dropped
            cancel();
            return;
        }
        System.err.println("Bark Bark");
        if (failCount > 120) {
            server.getPendingConnection(id).close();
            cancel();
            System.err.println("Growl");
            System.err.println("\n\n\n\n\n");
            return;
        }
        server.greeting(id);
        failCount++;
    }

}
